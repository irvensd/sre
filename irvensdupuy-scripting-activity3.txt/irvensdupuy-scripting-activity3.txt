# Create a file with the date in the name
date=$(date +%Y-%m-%d)
filename="$date.txt"

# Create a function to create the file
function createfile {
	touch $filename
}

# Create a function to generate the lottery numbers
function generatelottery {
	for i in {1..5}
	do
		lottery+=($((RANDOM%50+1)))
	done
	bonus=$((RANDOM%10+1))
}

# Create a function to write the lottery numbers to the file
function writetofile {
	for i in ${lottery[@]}
	do
		echo $i >> $filename
	done
	echo $bonus >> $filename
}

# Create a function to check the lottery numbers
function checklottery {
	for i in ${lottery[@]}
	do
		if [[ $i -eq $1 ]]
		then
			matches=$((matches+1))
		fi
	done
	if [[ $bonus -eq $1 ]]
	then
		matches=$((matches+1))
	fi
}

# Create a function to output the results
function outputresult {
	echo "The lottery numbers are:"
	for i in ${lottery[@]}
	do
		echo $i
	done
	echo "The bonus number is:"
	echo $bonus
	echo "You matched $matches numbers"
}

# Create a function to check if the file exists and if not create it
function checkfile {
	if [[ -f $filename ]]
	then
		echo "File exists"
	else
		createfile
	fi
}

# Create a function to run the program
function runprogram {
	checkfile
	generatelottery
	writetofile
	read -p "Enter a number to check: " number
	checklottery $number
	outputresult
}

# Run the program
runprogram
