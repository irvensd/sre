# Set the date in the filename
DATE=$(date +%Y%m%d)
# Create the file
echo "Creating file: $DATE.txt"
touch $DATE.txt
# Create the file if it doesn't exist
if [ ! -f $DATE.txt ]; then
    echo "Creating file: $DATE.txt"
    touch $DATE.txt
fi

# Set the number of draws
NUMBER_OF_DRAWS=5
# Set the number of bonus balls
NUMBER_OF_BONUS_BALLS=1
# Set the number of numbers in the set
NUMBER_OF_NUMBERS=50

# Loop through the draws
for (( i=1; i <= $NUMBER_OF_DRAWS; i++ )); do
    echo "Draw $i:" >> $DATE.txt
    # Loop through the numbers
    for (( j=1; j <= $NUMBER_OF_NUMBERS; j++ )); do
        # Pick a random number
        NUMBER=$(($RANDOM % $NUMBER_OF_NUMBERS + 1))
        # Check to see if the number is already in the set
        if [ -z $(grep $NUMBER $DATE.txt) ]; then
            # If not, add it to the set
            echo "$NUMBER" >> $DATE.txt
        else
            # If it is, pick a new number
            NUMBER=$(($RANDOM % $NUMBER_OF_NUMBERS + 1))
            echo "$NUMBER" >> $DATE.txt
        fi
    done
    # Pick a random number for the bonus ball
    NUMBER=$(($RANDOM % $NUMBER_OF_BONUS_BALLS + 1))

    fi
